mkcd() {
	mkdir $1
	pushd $1
}

cat_or_ls() { 
	if [ -f "$1" ]
	then
		less -RF $1
	else
		ls -al $1
	fi
}

push_or_pop() {	if [ "$1" = "" ]
	then
		popd
	else
		pushd $1
	fi
}
parse_git_branch() {
  # TODO: find a git command that will give me this directly, instead of needing sed
  git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/(\1)/'
}
